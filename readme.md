```javascript
var weekdays = ['Воскресенье', 'Понедельник', 'Вторник', 'Среда', 'Четверг', 'Пятница', 'Суббота'];

function getDay() {

  var day = date.getDay();

}
```

```javascript
getDay(); // Суббота
getDay(); // Strange day
```
